	B -->> K: 受注（受注商品付き）
	loop 受注数
		K ->> K: システム利用回数++
		loop 受注商品数
			opt キックバック対象商品
				K ->> B: 定価取得 /product_sets
				B -->> K: 定価
				K ->> K: キックバック計算
			end
		end
	end
end

K ->> Tran: キックバック計算結果
Tran -->> K: 保存OK
