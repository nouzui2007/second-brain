# Case Classからjson出力（実装）

## アルゴリズム

1. フィールド名を取得
2. 値を取得
3. zipで合成

```scala
case class User(name: String, age: Int)
val user = User("牧内", 48)
val fields = user.getClass.getDeclaredFields.map(_.getName)
val values = user.productIterator.toList

// {name: 牧内,age: 48}
fields.zip(values).map {
  case (field, value) => s"${field}: ${value}"
}.mkString("{", ",", "}")
```

## Companionオブジェクト

```scala
import play.api.libs.json.Json

// 会社
case class Company(name: String, address: String)
object Company {
  implicit val format = Json.format[Company]
}

// ユーザー
case class User(name: String, age: Int, language: List[String], company: Company)
object User {
  implicit val format = Json.format[User]
}

val jsonString = """
{
  "name": "makiuchi",
  "age": 48,
  "language": ["Scala", "C#", "Delphi"],
  "company": {
    "name": "ccube",
    "address": "東京都千代田区神田須田町"
  }
}
"""

val user = Json.parse(jsonString).validate[User]
user

val userString = Json.toJson(user.get)
userString
```

## JsSuccess、JsError

```scala
import play.api.libs.json.{Json, JsError, JsSuccess}

// 会社
case class Company(name: String, address: String)
object Company {
  implicit val format = Json.format[Company]
}

// ユーザー
case class User(name: String, age: Int, language: List[String], company: Company)
object User {
  implicit val format = Json.format[User]
}

val jsonString = """
{
  "name": "makiuchi",
  "age": 48,
  "language": ["Scala", "C#", "Delphi"],
  "company": {
    "name": "ccube",
    "address": "東京都千代田区神田須田町"
  }
}
"""

// エラーにも対応
Json.parse(jsonString).validate[User] match {
  case JsSuccess(value, _) => Json.toJson(value)
  case JsError(errors) => {
    errors.map { e =>
      println("Path: " + e._1.toString())
      println("内容：")
      e._2.map { e2 =>
        println(e2.message)
      }
    }    
  }
}
```
