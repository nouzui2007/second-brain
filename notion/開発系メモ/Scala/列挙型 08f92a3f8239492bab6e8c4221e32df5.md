# 列挙型

```scala
sealed abstract class Fruits(val code: Int) {
  val name = toString
}

object Fruits {
  case object Apple  extends Fruits(10)
  case object Orange extends Fruits(21)
  case object Banana extends Fruits(34)

  // 列挙メンバーのリストは自分で実装する必要がある
  lazy val values: IndexedSeq[Fruits] = 
    IndexedSeq(Fruits.Apple, Fruits.Orange, Fruits.Banana)
  
  def complement(code: Int) = values.find(_.code == code)
}

println(Fruits.values(1))
println(Fruits.values(1).code)
println(Fruits.complement(35).getOrElse("BAAAAAN!"))
```
